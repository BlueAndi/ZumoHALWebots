95 void clear() final
96 {
97 if (nullptr != m_display)
98 {
99 m_display->setColor(BLACK);
100 m_display->fillRectangle(0, 0, m_width, m_height);
101 }
102
103 gotoXY(0, 0);
104 }

112 void gotoXY(uint8_t xCoord, uint8_t yCoord) final
113 {
114 /* The one gets added to have a better central allignment of the text. */
115 m_currY = static_cast<uint8_t>((m_height / (DISPLAYLINES + 1)) * (1 + yCoord));
116 m_currX = static_cast<uint8_t>((m_width / DISPLAYCHARS) * xCoord);
117
118 m_lastX = xCoord;
119 m_lastY = yCoord;
120 }
121
129 size_t print(const char str[]) final
130 {
131 std::string text;
132 text = str;
133
134 if (nullptr != m_display)
135 {
136 m_display->setColor(WHITE);
137 m_display->drawText(text, m_currX, m_currY);
138 }
139
140 gotoXY((m_lastX + text.length()), m_lastY);
141
142 return text.length();
143 }
